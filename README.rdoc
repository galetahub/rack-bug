Usage:

 script/plugin install git://github.com/brynary/rack-bug.git 

 # config/environments/development.rb
 config.middleware.use "Rack::Bug"

 # add bookmarklet to browser
 open http://RAILS_APP/__rack_bug__/bookmarklet.html
